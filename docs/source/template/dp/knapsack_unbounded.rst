程式碼
************************

::

    dp[i][j] = 使用第 1 ~ i 種物件，在總重 j 的限制下的最大價值
    dp[0][0] = 0
    考慮第 i 種物件使用的個數
    dp[i][j] = max([
      dp[i - 1][j - k * w[i]] + k * v[i]) for 0 <= k <= (j // w[i])
    ])
    dp[i][j] = max(dp[i - 1][j], dp[i- 1][j - w[i]] + v[i])

.. code-block:: cpp
    :linenos:

    int N, W, w[i], v[i];
    int dp[MAX_W + 1];

    dp[0] = 0;
    for (int i = 1; i <= N; i++) {
        for (int j = w[i]; j <= W; j++) {
            dp[j] = max(dp[j], dp[j - w[i]] + v[i]);
        }
    }
