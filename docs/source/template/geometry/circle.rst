.. code-block:: cpp
    :linenos:

    // d = 圓心距, R, r 為半徑 (R >= r)
    // 內切
    d = R - r
    // 外切
    d = R + r
    // 內離
    d < R - r
    // 外離
    d > R + r
    // 相交
    d < R + r && d > R - r
